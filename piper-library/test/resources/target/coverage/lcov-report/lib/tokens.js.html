"use strict";
/**
 * convert a token from base64 to hex and vice versa
 */
function convertToken(inputToken) {
    var result = { base64: "", hex: "" };
    if (inputToken && inputToken.length === 44) {
        result.base64 = inputToken;
        result.hex = (new Buffer(inputToken, "base64")).toString("hex");
    }
    else if (inputToken.length === 64) {
        result.hex = inputToken;
        result.base64 = (new Buffer(inputToken, "hex")).toString("base64");
    }
    return result;
}
exports.convertToken = convertToken;
