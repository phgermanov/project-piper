/**  extract connection string from environment variables or use default for
    local */
function mongodb_uri_from_env(env) {
    // check cloudfoundry first
    var vcap_services = env["VCAP_SERVICES"];
    if (vcap_services) {
        var vcap_services_json = void 0;
        try {
            // cloudfoundry binding contains database name already
            vcap_services_json = JSON.parse(vcap_services);
        }
        catch (e) {
            winston.error("failed to parse VCAP_SERVICES json\n", vcap_services);
            return null;
        }
        return mongodb_uri_from_VCAP_SERVICES(vcap_services_json);
    }
    // check for docker variables
    var mongo_hostname = env["MONGO_PORT_27017_TCP_ADDR"] || "localhost";
    var mongo_port = env["MONGO_PORT_27017_TCP_PORT"] || "27017";
    var mongo_connection_string = "mongodb://" + mongo_hostname + ":" + mongo_port + "/" + mongo_database_name;
    return mongo_connection_string;
}
exports.mongodb_uri_from_env = mongodb_uri_from_env;
